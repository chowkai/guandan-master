## 微信小程序实现

### WXML 结构

```xml
<view class="game-screen">
  <!-- 顶部信息栏 -->
  <view class="top-bar">
    <view class="pause-btn" bindtap="pauseGame">⏸️</view>
    <view class="game-info">
      <text class="round">第{{round}}局</text>
      <text class="score">{{score}}</text>
    </view>
  </view>
  
  <!-- 对家信息 -->
  <view class="player-info opponent">
    <text>对家 AI</text>
    <text>(剩余{{opponentCards}}张)</text>
  </view>
  
  <!-- 出牌区域 -->
  <view class="play-area">
    <view class="played-cards">
      <view class="card" wx:for="{{playedCards}}">{{item}}</view>
    </view>
  </view>
  
  <!-- 手牌区域 -->
  <view class="hand-area">
    <scroll-view scroll-x>
      <view class="cards-container">
        <view class="card {{item.selected ? 'selected' : ''}}" 
              wx:for="{{handCards}}"
              bindtap="selectCard">
          {{item.value}}{{item.suit}}
        </view>
      </view>
    </scroll-view>
  </view>
  
  <!-- 操作按钮 -->
  <view class="action-buttons">
    <button class="btn" bindtap="hint">提示</button>
    <button class="btn {{canPlay ? 'active' : 'disabled'}}" 
            bindtap="playCards">出牌</button>
    <button class="btn" bindtap="undo">撤销</button>
  </view>
</view>
```
